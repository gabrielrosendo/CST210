(y/n)";            cout << "\n";            cin >> str;        }    }}void showDice(Die dice[]){    for (int i = 1; i <= 5; i++)    {        cout << i;        cout << " ";    }    cout << " " << endl;    for (int j = 0; j < 5; j++)    {        cout << "-";        cout << ' ';    }    cout << "\n";    // print out all values in the die[]    for (int i = 0; i < 5; i++)    {        cout << dice[i].getValue();        cout << " ";    }    cout << "\n";}int getTotalScore(Die dice[]){    // initialize total to 0    int total = 0;    for (int i = 0; i < 5; i++)    {        // add value of each int to total        total += dice[i].getValue();    }    return total;}
